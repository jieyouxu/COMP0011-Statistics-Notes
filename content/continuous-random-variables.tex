\subsection{Continuous Random Variables}

\begin{definition}[Cumulative Distribution Function]
    For the random variable $X$, is \textit{cumulative distribution function}
    $F(X)$ is the function

    \begin{equation}
        \Forall x \in \Real \colon F(X) = P(X \le x)
    \end{equation}
\end{definition}
